; =======================================================
; Comparación CMPI (registro con inmediato)
; Resultado esperado: Flags Z=1, N=0, C=1, V=0 (son iguales)
; =======================================================
MOVI REG1, #16             ; REG1 = 16
MOVI REG2, #16             ; REG2 = 16
CMPI REG1, #16             ; Compara REG1 con 16: son iguales → Z=1
SWI                        ; Estado final: REG1=16, REG2=16, Flags: Z=1
