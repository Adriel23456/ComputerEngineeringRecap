; =======================================================
; Floating-point absolute value with immediate
; Expected result: REG3 = 15.0 (|-15.0| = 15.0)
; =======================================================
FMOVI REG1, #1.0           ; REG1 = 1.0
FMOVI REG2, #16.0          ; REG2 = 16.0
FABSI REG3, #-15.0         ; REG3 = |-15.0| = 15.0
NOP
NOP
NOP
SWI                        ; Final state: REG1 = 1.0, REG2 = 16.0, REG3 = 15.0
